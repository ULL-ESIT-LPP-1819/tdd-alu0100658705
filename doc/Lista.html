def push_head(value)
	if @head == nil
		@head = @Node.new(value, nil, nil)
		@tail = @head
	else
		@head.prev=@Node.new(value, @head, nil)
		@head = @head.prev
	end
end

def push_tail(value)
	if @head == nil
		@head = @Node.new(value, nil, nil)
	   	@tail = @head
	else
		@tail.next = @Node.new(value, nil, @tail)
		@tail = @tail.next
	end
end
