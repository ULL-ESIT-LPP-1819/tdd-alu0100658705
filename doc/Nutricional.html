def Formatear(a,b)
        if ( a == grasa)
                porc=((a*100*37)/b).round(2)
                calorias=37*a
        end
        
               if ( a == hidratos)
                porc=((a*100*17)/b).round(2)
                calorias=17*a
        end

        if ( a == proteinas)
                porc=((a*100*17)/b).round(2)
                calorias=17*a
        end
        
        if (a == sal)
                porc=((a*100*25)/b).round(2)
                calorias=25*a
        end

        return porc, calorias
end
